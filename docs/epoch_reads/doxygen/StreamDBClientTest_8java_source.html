123 private LogfileRecord logfileRecordForEpoch(long epoch, boolean hasNullEpochColumns) {
124 Instant instant = Instant.ofEpochSecond(epoch);
125 ZonedDateTime zonedDateTime = instant.atZone(zoneId); // expects path dates to be in same timezone as mariadb system timezone
126 int year = zonedDateTime.getYear();
127 // format 0 in front of string if 1-9
128 String month = String.format("%02d", zonedDateTime.getMonthValue());
129 String day = String.format("%02d", zonedDateTime.getDayOfMonth());
130 String hour = String.format("%02d", zonedDateTime.getHour());
131
132 String filename = "example.log-@" + epoch + "-" + year + month + day + hour + ".log.gz";
133 String path = year + "/" + month + "-" + day + "/example.tg.dev.test/example/" + filename;
134 LogfileRecord logfileRecord = new LogfileRecord(
135 ULong.valueOf(ThreadLocalRandom.current().nextLong(0L, Long.MAX_VALUE)),
136 Date.valueOf(zonedDateTime.toLocalDate()),
137 Date.valueOf(zonedDateTime.plusYears(1).toLocalDate()),
138 UShort.valueOf(1),
139 path,
140 null,
141 UShort.valueOf(1),
142 filename,
143 new Timestamp(epoch),
144 ULong.valueOf(120L),
145 "sha256 checksum 1",
146 "archive tag 1",
147 "oldExample",
148 UShort.valueOf(2),
149 UShort.valueOf(1),
150 ULong.valueOf(390L),
151 ULong.valueOf(epoch),
152 ULong.valueOf(epoch + (365 * 24 * 3600)),
153 ULong.valueOf(epoch),
154 ULong.valueOf(1),
155 null
156 );
157
158 LogfileRecord nullEpochRecord = new LogfileRecord(
159 ULong.valueOf(ThreadLocalRandom.current().nextLong(0L, Long.MAX_VALUE)),
160 Date.valueOf(zonedDateTime.toLocalDate()),
161 Date.valueOf(zonedDateTime.plusYears(1).toLocalDate()),
162 UShort.valueOf(1),
163 path,
164 null,
165 UShort.valueOf(1),
166 filename,
167 new Timestamp(epoch),
168 ULong.valueOf(120L),
169 "sha256 checksum 1",
170 "archive tag 1",
171 "oldExample",
172 UShort.valueOf(2),
173 UShort.valueOf(1),
174 ULong.valueOf(390L),
175 null,
176 null,
177 null,

179 null
